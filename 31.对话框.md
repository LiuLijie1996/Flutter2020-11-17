## AlertDialog 对话框

```dart
class Page1 extends StatelessWidget {
  const Page1({Key key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("页面1"),
      ),
      body: Column(
        children: [
          RaisedButton(
            onPressed: () async {
              // 弹出对话框
              bool result = await showDialogFunc(context);
              if (result == true) {
                print("点击的是确定：$result");
              } else if (result == false) {
                print("点击的是取消：$result");
              } else {
                print("点击的是遮罩层：$result");
              }
            },
            child: Text("点我弹出对话框"),
          )
        ],
      ),
    );
  }

  Future<bool> showDialogFunc(BuildContext context) {
    // 使用对话框时得借助 showDialog 进行构建
    return showDialog(
      context: context,
      builder: (context) {
        return AlertDialog(
          title: Text("弹出提示"),
          actions: [
            FlatButton(
              onPressed: () {
                // 关闭对话框，第二个参数是传入的数据
                Navigator.pop(context, false);
              },
              child: Text("取消"),
            ),
            FlatButton(
              onPressed: () {
                // 关闭对话框，第二个参数是传入的数据
                Navigator.pop(context, true);
              },
              child: Text("确定"),
            ),
          ],
        );
      },
    );
  }
}
```



## SimpleDialog 对话框

```dart
class Page2 extends StatelessWidget {
  const Page2({Key key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("页面2"),
      ),
      body: Column(
        children: [
          RaisedButton(
            onPressed: () async {
              // 弹窗
              int result = await showDialogFunc(context);
              switch (result) {
                case 0:
                  print("简体中文");
                  break;
                case 1:
                  print("繁体中文");
                  break;
                case 2:
                  print("英文");
                  break;
                default:
              }
            },
            child: Text("点我弹窗"),
          )
        ],
      ),
    );
  }

  Future<int> showDialogFunc(BuildContext context) {
    // 使用弹窗时得借助 showDialog 进行构建
    return showDialog(
      context: context,
      builder: (context) {
        return SimpleDialog(
          title: Text("请选择语言"),
          children: [
            FlatButton(
              onPressed: () {
                // 关闭弹窗，第二个参数是传入的数据
                Navigator.pop(context, 0);
              },
              child: Text("简体中文"),
            ),
            FlatButton(
              onPressed: () {
                // 关闭弹窗，第二个参数是传入的数据
                Navigator.pop(context, 1);
              },
              child: Text("繁体中文"),
            ),
            FlatButton(
              onPressed: () {
                // 关闭弹窗，第二个参数是传入的数据
                Navigator.pop(context, 2);
              },
              child: Text("US English"),
            ),
          ],
        );
      },
    );
  }
}
```



## Dialog

```dart
class Page3 extends StatelessWidget {
  const Page3({Key key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("页面3"),
      ),
      body: Column(
        children: [
          RaisedButton(
            onPressed: () async {
              // 弹窗
              int result = await showDialogFunc(context);
              print("你点击了第 $result 项");
            },
            child: Text("点我弹窗"),
          )
        ],
      ),
    );
  }

  Future<int> showDialogFunc(BuildContext context) {
    // 使用弹窗时得借助 showDialog 进行构建
    return showDialog(
      context: context,
      builder: (context) {
        return Dialog(
          child: Column(
            children: [
              ListTile(
                title: Text("请选择"),
              ),
              Expanded(
                child: ListView.builder(
                  itemBuilder: (BuildContext context, int index) {
                    return FlatButton(
                      onPressed: () {
                        Navigator.pop(context, index);
                      },
                      child: ListTile(
                        title: Text("index：$index"),
                      ),
                    );
                  },
                  itemCount: 30,
                ),
              ),
            ],
          ),
        );
      },
    );
  }
}
```



## showDialog  弹窗构建器

```dart
class Page4 extends StatelessWidget {
  const Page4({Key key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("页面4"),
      ),
      body: Column(
        children: [
          RaisedButton(
            onPressed: () {
              // 弹窗
              showDialogFunc(context);
            },
            child: Text("点我弹窗"),
          )
        ],
      ),
    );
  }

  Future showDialogFunc(BuildContext context) {
    // 使用弹窗时得借助 showDialog 进行构建
    return showDialog(
      context: context,
      builder: (context) {
        // 返回的内容可以不是 Dialog 类型的数据，也可以是别的组件
        return FlatButton(
          onPressed: () {
            Navigator.pop(context);
          },
          child: Container(
            width: 100.0,
            height: 100.0,
            color: Colors.white,
            alignment: Alignment.center,
            child: Text("我不是Dialog"),
          ),
        );
      },
    );
  }
}
```
