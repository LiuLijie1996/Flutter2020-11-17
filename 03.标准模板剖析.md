## 计算器示例解析

### 第一部分：应用入口、应用结构、页面结构

```dart
import 'package:flutter/material.dart';

//dart入口函数
void main() {
  //程序入口函数
  runApp(MyApp());
}

// MyApp属于基类，继承了 StatelessWidget， 表示这是一个无状态的 widget，其状态从始至终不会发生改变
class MyApp extends StatelessWidget {
  // build 用来构建 widget 的方法
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      ...
      // 指定一个页面入口
      home: MyHomePage(title: 'hello flutter'),
    );
  }
}

// 该类继承 StatefulWidget 表示这是一个有状态的 widget
class MyHomePage extends StatefulWidget {
  // 该构造函数接收一个 title 属性

  final String title;

  @override
  _MyHomePageState createState() => _MyHomePageState();
}
// 该类继承 State 表示内部的数据是不定的，需要实时进行更新的
class _MyHomePageState extends State<MyHomePage> {
  @override
  Widget build(BuildContext context) {...}
}
```

### 第二部分：状态管理

```dart
// StatefulWidget 表示是一个有状态的组件
class MyHomePage extends StatefulWidget {
  // 这是一个组件的构造函数
  MyHomePage({Key key, this.title}) : super(key: key);

  final String title;

  @override
  _MyHomePageState createState() => _MyHomePageState();
}

// 在定义一个有状态的组件后，需要继承State表示页面数据是不定的，需要实时更新
class _MyHomePageState extends State<MyHomePage> {
  int _counter = 0;

  //点击悬浮按钮后触发该函数
  void _incrementCounter() {
    //调用setState函数修改_counter，更新页面
    setState(() {
      _counter++;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(widget.title),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: <Widget>[
            Text(
              'You have pushed the button this many times:',
            ),
            Text(
              '$_counter',
              style: Theme.of(context).textTheme.headline4,
            ),
          ],
        ),
      ),
      //悬浮按钮
      floatingActionButton: FloatingActionButton(
        //点击后触发 _incrementCounter 事件函数
        onPressed: _incrementCounter,
        tooltip: 'Increment',
        child: Icon(Icons.add),
      ),
    );
  }
}
```
